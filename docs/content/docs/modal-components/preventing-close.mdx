title: Preventing close

Modals can prevent themselves from closing by using `onBeforeClose`. 
If the provided callback returns `false` the modal will not close.

```svelte
<script>
  import { onBeforeClose } from 'svelte-modals'

  const { isOpen, close } = $props()

  let canClose = $state(true)

  onBeforeClose(() => {    
    return canClose
  })
</script>
```


```svelte live layout="$lib/layouts/code/NoCode.astro"
<script>
  import { modals } from 'svelte-modals'
  import PreventCloseModal from '$lib/components/PreventCloseModal.svelte'

  function onclick() {
    modals.open(PreventCloseModal)
  }
</script>

<button {onclick}>Open</button>
```
